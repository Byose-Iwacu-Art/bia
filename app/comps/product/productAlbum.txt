import { SetStateAction, useState } from "react";

export default function ProductAlbum({ product }) {
    const [currentImageIndex, setCurrentImageIndex] = useState(0);

    const handleImageClick = (index: any) => {
        setCurrentImageIndex(index);
    };

    const handleNext = () => {
        setCurrentImageIndex((prevIndex) =>
            prevIndex === product.images.length - 1 ? 0 : prevIndex + 1
        );
    };

    const handlePrevious = () => {
        setCurrentImageIndex((prevIndex) =>
            prevIndex === 0 ? product.images.length - 1 : prevIndex - 1
        );
    };

    return (
        <div className="album flex flex-col">
            <div className="main-pic relative p-4 h-[60vh] mx-4 my-3 w-full shadow-sm rounded-md bg-zinc-100">
                <img
                    src={product.images[currentImageIndex]}
                    alt=""
                    className="w-full h-full object-contain"
                />
                <div className="controls absolute top-1/2 left-0 right-0 flex justify-between px-4">
                    <button
                        onClick={handlePrevious}
                        className="bg-black bg-opacity-50 text-white p-2 rounded-full"
                    >
                        <i className="bi bi-chevron-left"></i>
                    </button>
                    <button
                        onClick={handleNext}
                        className="bg-black bg-opacity-50 text-white p-2 rounded-full"
                    >
                        <i className="bi bi-chevron-right"></i>
                    </button>
                </div>
            </div>
            <div className="other p-2 flex flex-row">
                {product.images.map((src, index) => (
                    <div
                        key={index}
                        onClick={() => handleImageClick(index)}
                        className={`pic w-[100px] h-[80px] p-1 rounded-md bg-zinc-100 mx-1 cursor-pointer ${
                            currentImageIndex === index
                                ? "border-2 border-blue-500"
                                : ""
                        }`}
                    >
                        <img src={src} alt="" className="w-full h-full object-contain" />
                    </div>
                ))}
            </div>
        </div>
    );
}
